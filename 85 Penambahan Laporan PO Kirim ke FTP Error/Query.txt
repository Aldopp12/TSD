SELECT A.*, b.ftnama FROM (
SELECT NO_PO, ZIPFILE, SUPPLIER, UNIT, WILAYAH, CABANG, FTP_FROM, FTP_TO, KETERANGAN, TGL_UPD, JAM, KD_USER, 'P' DATA_ID FROM LOG_FTP_PO_TEMP --(CHECK)
UNION ALL
SELECT NO_PO, ZIPFILE, SUPPLIER, UNIT, WILAYAH, CABANG, FTP_FROM, FTP_TO, KETERANGAN, Tgl_upd, JAM, KD_USER, 'NP' DATA_ID FROM LOG_FTP_PO --(DITAMPILIN)
) A INNER JOIN T_SUPPLIER b ON A.SUPPLIER=b.ftkode WHERE DATA_ID='NP'
